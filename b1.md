**Mã Java mẫu:**

```java
import java.math.BigDecimal;
import java.math.RoundingMode;

public class PersonalIncomeTaxCalculator {

    private static final BigDecimal PERSONAL_DEDUCTION = new BigDecimal("11000000");
    private static final BigDecimal DEPENDENT_DEDUCTION = new BigDecimal("4400000");
    private static final BigDecimal INSURANCE_RATE = new BigDecimal("0.105");

    private static final BigDecimal[] BRACKETS = {
        new BigDecimal("5000000"),
        new BigDecimal("10000000"),
        new BigDecimal("18000000"),
        new BigDecimal("32000000"),
        new BigDecimal("52000000"),
        new BigDecimal("80000000")
    };

    private static final BigDecimal[] RATES = {
        new BigDecimal("0.05"),
        new BigDecimal("0.10"),
        new BigDecimal("0.15"),
        new BigDecimal("0.20"),
        new BigDecimal("0.25"),
        new BigDecimal("0.30"),
        new BigDecimal("0.35")
    };

    public BigDecimal calculate(BigDecimal grossIncome, int dependents, BigDecimal insuranceAmount) {
        BigDecimal insurance = insuranceAmount != null ? insuranceAmount : grossIncome.multiply(INSURANCE_RATE);
        BigDecimal taxableIncome = grossIncome
                .subtract(PERSONAL_DEDUCTION)
                .subtract(DEPENDENT_DEDUCTION.multiply(BigDecimal.valueOf(dependents)))
                .subtract(insurance);

        if (taxableIncome.compareTo(BigDecimal.ZERO) <= 0) {
            return BigDecimal.ZERO.setScale(0, RoundingMode.HALF_UP);
        }

        BigDecimal tax = BigDecimal.ZERO;
        BigDecimal lowerBound = BigDecimal.ZERO;

        for (int i = 0; i < RATES.length; i++) {
            BigDecimal upperBound = i < BRACKETS.length ? BRACKETS[i] : taxableIncome;
            BigDecimal taxableAtThisBand = taxableIncome.min(upperBound).subtract(lowerBound);

            if (taxableAtThisBand.compareTo(BigDecimal.ZERO) > 0) {
                tax = tax.add(taxableAtThisBand.multiply(RATES[i]));
            }

            if (taxableIncome.compareTo(upperBound) <= 0) {
                break;
            }
            lowerBound = upperBound;
        }

        return tax.setScale(0, RoundingMode.HALF_UP);
    }
}
```
